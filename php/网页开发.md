function convertUnderline($str, $ucfirst = false)
{
	$str = explode('_', $str);

	foreach ($str as $key => $val)
		$str[$key] = ucfirst($val);

	if (!$ucfirst)
		$str[0] = strtolower($str[0]);

	return implode('', $str);
}
